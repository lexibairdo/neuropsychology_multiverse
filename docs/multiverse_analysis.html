Generate a function that computes the Cohenâ€™s F2 value and updates the dataframe with a column that has these values

add_f2 <- function(data, r2_column = "fit_r.squared") {
  if (!r2_column %in% names(data)) {
    stop("Specified R2 column not found in data.")
  }
  
  r2_col <- sym(r2_column)
  
  result <- data |>
    mutate(
      f2_value = !!r2_col / (1 - !!r2_col),
      effect_size = case_when(
        f2_value <= 0.14             ~ "small",
        f2_value > 0.14 & f2_value <= 0.34 ~ "medium",
        f2_value > 0.34              ~ "large"
      )
    )
  
  # Print summary table
  result |>
    count(effect_size) |>
    print()
  
  # Calculate FÂ² ratio
  f2_max <- max(result$f2_value, na.rm = TRUE)
  f2_min <- min(result$f2_value, na.rm = TRUE)
  f2_ratio <- round(f2_max / f2_min, 2)
  
  # Generate barplot with ratio displayed in the caption
  p <- ggplot(result, aes(x = effect_size, fill = effect_size)) +
    geom_bar() +
    scale_fill_manual(
      values = c(
        "large" = "darksalmon",
        "medium" = "cyan4",
        "small" = "darkgoldenrod1"
      )
    ) +
    labs(
      title = "Counts of Cohen's fÂ² Effect Sizes",
      subtitle = paste("FÂ² max/min ratio =", f2_ratio),
      x = "Effect Size Category",
      y = "Count"
    ) +
    theme_minimal()
  
  print(p)
  
  return(result)
}
